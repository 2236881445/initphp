-- TableName initpm_sql_table 项目管理平台 - 数据表信息
-- Created By initphp.com@2012-01-01
-- Fields id          		主键ID
-- Fields project_id        项目ID
-- Fields table_name_en  	英文名
-- Fields table_name_cn  	中文名
-- Fields table_name_des  	表名描述
-- Fields table_struct  	表字段结构
-- Fields table_code  	    表设计代码
CREATE TABLE `initpm_sql_table` (  
	`id` int(10) unsigned NOT NULL AUTO_INCREMENT, 
	`project_id` int(10) unsigned NOT NULL DEFAULT '0', 
	`table_name_en` varchar(30) NOT NULL DEFAULT '', 
	`table_name_cn` varchar(30) NOT NULL DEFAULT '', 
	`table_name_des` varchar(100) NOT NULL DEFAULT '', 
	`table_struct` text NOT NULL,
	`table_code` text NOT NULL,
	 PRIMARY KEY (`id`),    
	 KEY `idx_table_name_en` (`table_name_en`),
	 KEY `idx_table_name_cn` (`table_name_cn`),
	 KEY `idx_project_id` (`project_id`)
) ENGINE=INNODB DEFAULT CHARSET=utf8;


-- TableName initpm_project_team 项目管理平台 - 项目组
-- Created By initphp.com@2012-01-01
-- Fields id          		主键ID
-- Fields team_name         项目组名称
-- Fields team_des          项目组描述
-- Fields create_time       创建时间
CREATE TABLE `initpm_project_team` (  
	`id` int(10) unsigned NOT NULL AUTO_INCREMENT, 
	`team_name` varchar(30) NOT NULL DEFAULT '', 
	`team_des` varchar(255) NOT NULL DEFAULT '', 
	`create_time` int(10) unsigned NOT NULL DEFAULT '0', 
	 PRIMARY KEY (`id`)    
) ENGINE=INNODB DEFAULT CHARSET=utf8;

-- TableName initpm_project 项目管理平台 - 团队项目
-- Created By initphp.com@2012-01-01
-- Fields id          		主键ID
-- Fields project_name      项目名称
-- Fields project_des       项目描述
-- Fields create_time       创建时间
CREATE TABLE `initpm_project` (  
	`id` int(10) unsigned NOT NULL AUTO_INCREMENT, 
	`team_id` int(10) unsigned NOT NULL DEFAULT '0', 
	`project_name` varchar(30) NOT NULL DEFAULT '', 
	`project_des` varchar(255) NOT NULL DEFAULT '', 
	`create_time` int(10) unsigned NOT NULL DEFAULT '0', 
	 PRIMARY KEY (`id`)    
) ENGINE=INNODB DEFAULT CHARSET=utf8;
